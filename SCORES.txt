(
    player_name varchar(30),
    gender varchar(1),
    DAYSS date,
    score_points int

#COMMAND AS PER REQUIRED CONDITIONS

select s.gender, s.DAYSS, (select sum(score_points) from scores where gender = s.gender and DAYSS <= s.DAYSS) as total
    from scores s
    group by gender, DAYSS
    order by gender,DAYSS;
